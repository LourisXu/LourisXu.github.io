已知入栈顺序，计算出栈可能性
动规：f(n,m)表示n个待入栈，m个已入栈
f(n,m)=f(n-1,m+1)+f(n,m-1)表示当前状态只会出现两种情况：
①一个进栈
②一个出栈
二者取其一
结束条件：
n=0,f=1;
m=0,f(n,m)=f(n-1,1); //只能入栈，不能出栈

const int MaxSize=20;
typdef int ElemType;
typedef struct ComStack{
	ElemType stack[MaxSize];
	int top[2];
}ComStack;

bool push(ComStack s,int i, ElemType x){
	//左栈栈顶top0，右栈栈顶top1
	//top0初始-1，top1初始MaxSize
	if(s.top[1]-s.top[0]==1){
		return false; //栈满返回
	}
	if(i!=0||i!=1){
		return false;  //栈号错误
	}
	switch(i){
		case 0:
			s.stack[++top[0]]=x;
			break;
		case 1:
			s.stack[--top[1]]=x;
			break;
		default:break;
	}
	return true;
}

bool pop(ComStack s,int i, ElemType &e){
	if(i!=0||i!=1){
		return false;  //栈号错误
	}
	if(s.top[i]==-1||s.top[i]==MaxSize){
		return false;  //栈空，无法pop
	}
	switch(i){
		case 0:
			e=s.stack[s.top[0]--];
			break;
		case 1:
			e=s.stack[s.top[0]++];
			break;
	}
	return true;
}

int balanceBinaryTree(BiTree T,bool &flag){
	if(!T){
		return 0;
	}else{
		int left=balanceBinaryTree(T->lchild,flag);
		int right=balanceBinaryTree(T->rchild,flag);

			flag=false;
		}
		return max(left,right)+1;
	}
}

		return true;

		balanceBinaryTree(T->lchild,h);
		int left=h;
		balanceBinaryTree(T->rchild,h);
		int right=h;

		h=max(left,right)+1;
		if(abs(left-right)>1){
			return false;
		}else{
			return true;
		}
	}
}

void PreOrderTraverse(BiTree T){

	InitStack(s);
	BiTree p=T;
	whie(p||!StackEmpty(s)){
		if(p){
			visit(p);
			if(p->rchild){
				Push(s,p->rchild);
			}
			p=p->lchild;
		}else{
			Pop(s,p);
		}
	}
}

void InOrderTraverse(BiTree T){

	InitStack(s);
	BiTree p=T;
	while(p||!StackEmpty(s)){
		if(p){
			Push(s,p);
			p=p->lchild;
		}else{
			Pop(s,p);
			visit(p);
			p=p->rchild;
		}
	}
}

void PostOrderTraverse(BiTree T){

	InitStack(s);
	BiTree p=T;
	BiTree pre=NULL;
	Push(s,p);
	while(p||!StackEmpty(s)){
		GetTop(s,p);
		if((!p->lchild&&!p->rchild)||(pre!=NULL&&(pre==p->lchild||pre==p->rchild))){
			visit(p);
			pre=p;
			Pop(s,p);
		}else{
			if(p->rchild){
				Push(s,p->rchild);
			}
			if(p->lchild){
				Push(s,p->lchild);
			}
		}
	}
}
